Função de Preload Otimizada function realizarPreload() { const statusTxt = document.getElementById('status-txt'); Object.keys(configuracao).forEach(pasta => { cacheImagens[pasta] = []; for (let i = 1; i <= configuracao[pasta]; i++) { const img = new Image(); const frame = String(i).padStart(3, '0'); // Monta o caminho - Verifique se a extensão é .png ou .PNG const caminho = `${pasta}/quadro${frame}.png`; img.onload = () => { imagensCarregadas++; atualizarProgresso(); }; img.onerror = () => { console.error("Erro ao carregar:", caminho); imagensCarregadas++; // Conta mesmo com erro para não travar a barra atualizarProgresso(); }; img.src = caminho; cacheImagens[pasta].push(img.src); } }); } function atualizarProgresso() { const percent = Math.floor((imagensCarregadas / totalImagens) * 100); const bar = document.getElementById('bar'); const statusTxt = document.getElementById('status-txt'); if (bar) bar.style.width = percent + '%'; if (statusTxt) statusTxt.innerText = `CARREGANDO: ${percent}%`; if (imagensCarregadas >= totalImagens) { console.log("Preload completo (com ou sem erros de imagem)."); setTimeout(() => { document.getElementById('loader').style.display = 'none'; document.querySelectorAll('.btn-smt').forEach(b => b.disabled = false); iniciarInstalacaoBase(); }, 500); } }
